TABLE TB_CUSTOMERS (
  customer_id INT PRIMARY KEY AUTO_INCREMENT,
  customer_name VARCHAR(100)
)

TABLE TB_PRODUCTS (
  product_id INT PRIMARY KEY AUTO_INCREMENT,
  product_name VARCHAR(100),
  price DECIMAL(10, 2)
)

TABLE TB_ORDER_STATUS (
  status_id INT PRIMARY KEY AUTO_INCREMENT,
  status_name VARCHAR(50)
)

TABLE TB_ORDERS (
  order_id INT PRIMARY KEY,
  customer_id INT REFERENCES TB_CUSTOMERS(customer_id),
  order_date DATETIME,
  status_id INT REFERENCES TB_ORDER_STATUS(status_id)
)

TABLE TB_ORDER_ITEMS (
  order_id INT REFERENCES TB_ORDERS(order_id),
  product_id INT REFERENCES TB_PRODUCTS(product_id),
  quantity INT
)
